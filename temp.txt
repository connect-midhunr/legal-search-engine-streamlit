    function generatecitation(pdfpaths,rootpath,citationnumb,file_name,isMob)
    {
        $.ajax({
                    type: "POST",
                    url: "https://hckinfo.kerala.gov.in/digicourt/Casedetailssearch/Generatecitationz",
                    data:'pdfpaths='+pdfpaths+'&rootpath='+rootpath+'&citationnumb='+citationnumb,
                    cache: false,
                    success: function(data)
                    {
                        //downloadpdf(file_name);
                        downloadPDForder(file_name)
                    }
                });
    }

    function downloadPDForder(file_name) {
        // Assuming the PHP variable $pdfFilePath is converted to a JavaScript variable
        var pdfFilePath = 'https://hckinfo.kerala.gov.in/digicourt//citation/'+file_name;
        // Create a temporary link element
        var link = document.createElement('a');
